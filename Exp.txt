local function IsEnemy(player)
    if not player or player == LocalPlayer then
        return false
    end

    if not TeamCheckEnabled then
        return true
    end

    local myTeam = LocalPlayer.Team
    local theirTeam = player.Team

    if not myTeam or not theirTeam then
        return true
    end

    if myTeam ~= theirTeam then
        return true
    end

    if LocalPlayer.TeamColor and player.TeamColor then
        if LocalPlayer.TeamColor ~= player.TeamColor then
            return true
        end
    end
    return false
end

local function RemoveESP(character)
    if ESPData[character] then
        for _, adornment in pairs(ESPData[character].Parts or {}) do
            if adornment then adornment:Destroy() end
        end
        if ESPData[character].Tag then
            ESPData[character].Tag:Destroy()
        end
        ESPData[character] = nil
    end
end

local function CreateESP(character, player)
    if ESPData[character] or not character.Parent then
        return
    end

    local humanoid = character:FindFirstChildOfClass("Humanoid")
    if not humanoid then
        return
    end

    local isEnemy = IsEnemy(player)
    local limbNames = {"Head", "HumanoidRootPart", "Left Arm", "Right Arm", "Left Leg", "Right Leg"}
    local limbParts = {}

    for _, limbName in ipairs(limbNames) do
        local limb = character:FindFirstChild(limbName)
        if limb then
            local box = Instance.new("BoxHandleAdornment")
            box.Name = "ESP_" .. limbName .. "_" .. player.Name
            box.Adornee = limb
            if limbName == "HumanoidRootPart" then
                box.Size = limb.Size + Vector3.new(0.2, 0.2, 0.2)
            else
                box.Size = limb.Size + Vector3.new(0.1, 0.1, 0.1)
            end
            box.Transparency = 0.6
            box.Color3 = isEnemy and Color3.fromRGB(255, 50, 50) or Color3.fromRGB(50, 255, 50)
            box.AlwaysOnTop = true
            box.ZIndex = 10
            box.Parent = ESPFolder
            table.insert(limbParts, box)
        end
    end

    local hrp = character:FindFirstChild("HumanoidRootPart")
    local billboard = nil
    local nameLabel = nil
    local infoLabel = nil

    if hrp then
        billboard = Instance.new("BillboardGui")
        billboard.Name = "ESP_Tag_" .. player.Name
        billboard.Adornee = hrp
        billboard.Size = UDim2.new(0, 200, 0, 60)
        billboard.StudsOffset = Vector3.new(0, 3, 0)
        billboard.AlwaysOnTop = true
        billboard.Parent = ESPFolder

        nameLabel = Instance.new("TextLabel")
        nameLabel.Size = UDim2.new(1, 0, 0.5, 0)
        nameLabel.BackgroundTransparency = 1
        nameLabel.Text = player.Name
        nameLabel.TextColor3 = isEnemy and Color3.fromRGB(255, 100, 100) or Color3.fromRGB(100, 255, 100)
        nameLabel.Font = Enum.Font.SourceSansBold
        nameLabel.TextSize = 14
        nameLabel.TextStrokeTransparency = 0
        nameLabel.TextStrokeColor3 = Color3.new(0, 0, 0)
        nameLabel.Parent = billboard

        infoLabel = Instance.new("TextLabel")
        infoLabel.Size = UDim2.new(1, 0, 0.5, 0)
        infoLabel.Position = UDim2.new(0, 0, 0.5, 0)
        infoLabel.BackgroundTransparency = 1
        infoLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
        infoLabel.Font = Enum.Font.SourceSans
        infoLabel.TextSize = 12
        infoLabel.TextStrokeTransparency = 0
        infoLabel.TextStrokeColor3 = Color3.new(0, 0, 0)
        infoLabel.Parent = billboard
    end

    ESPData[character] = {
        Parts = limbParts,
        Tag = billboard,
        NameLabel = nameLabel,
        InfoLabel = infoLabel,
        Player = player,
        IsEnemy = isEnemy,
        Connection = nil
    }
end

local function UpdateESP()
    if not ESPEnabled then
        return
    end
    for character, data in pairs(ESPData) do
        if not character.Parent then
            RemoveESP(character)
            continue
        end
        local player = data.Player
        if not player.Parent then
            RemoveESP(character)
            continue
        end
        local humanoid = character:FindFirstChildOfClass("Humanoid")
        if not humanoid or humanoid.Health <= 0 then
            RemoveESP(character)
            continue
        end
        local isEnemy = IsEnemy(player)
        if data.IsEnemy ~= isEnemy then
            data.IsEnemy = isEnemy
            for _, part in pairs(data.Parts or {}) do
                part.Color3 = isEnemy and Color3.fromRGB(255, 50, 50) or Color3.fromRGB(50, 255, 50)
            end
            if data.NameLabel then
                data.NameLabel.TextColor3 = isEnemy and Color3.fromRGB(255, 100, 100) or Color3.fromRGB(100, 255, 100)
            end
        end
        if data.InfoLabel and character:FindFirstChild("HumanoidRootPart") and LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
            local distance = (character.HumanoidRootPart.Position - LocalPlayer.Character.HumanoidRootPart.Position).Magnitude
            data.InfoLabel.Text = "HP: " .. math.floor(humanoid.Health) .. " | Dist: " .. math.floor(distance)
        end
    end
end

local function SetupPlayer(player)
    if player == LocalPlayer then
        return
    end
    if PlayerConnections[player] then
        return
    end
    local function OnCharacterAdded(char)
        char:WaitForChild("HumanoidRootPart", 5)
        char:WaitForChild("Humanoid", 5)
        if ESPEnabled then
            CreateESP(char, player)
        end
        local connection
        connection = char.AncestryChanged:Connect(function()
            if not char.Parent then
                RemoveESP(char)
                if connection then
                    connection:Disconnect()
                end
            end
        end)
        if ESPData[char] then
            ESPData[char].Connection = connection
        end
    end
    if player.Character then
        OnCharacterAdded(player.Character)
    end
    local charConnection = player.CharacterAdded:Connect(OnCharacterAdded)
    PlayerConnections[player] = charConnection
end

local function RemovePlayer(player)
    if PlayerConnections[player] then
        PlayerConnections[player]:Disconnect()
        PlayerConnections[player] = nil
    end
    if player.Character then
        RemoveESP(player.Character)
    end
end
